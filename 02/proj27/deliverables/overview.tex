\section{Exercise Description}

\begin{mdframed}[backgroundcolor=lightgray, linecolor=bordergray, linewidth=1pt,
    innerleftmargin=12pt, innerrightmargin=12pt,
    innertopmargin=10pt, innerbottommargin=10pt]
\textbf{8.9} \quad Write rules that will take a finite state machine from its present
state to its next state given a fact of the form

\vspace{0.5em}
\texttt{(input \textless value\textgreater)}

\vspace{0.5em}
The state machine and its arcs should be represented as facts. The input fact
should be retracted when the machine goes to its next state. Test your rules
and fact representations on the finite state machines shown in Figures~3.5
and~3.6.
\end{mdframed}

\subsection*{Approach}

The solution uses a \textbf{generic FSM engine} implemented as two CLIPS rules:

\begin{itemize}
  \item \texttt{fsm-transition} --- fires when a matching arc exists for the current
        state and input; retracts both the current state and the input fact, then
        asserts the new state.
  \item \texttt{fsm-no-transition} --- fires when no arc matches; prints an error
        message.
\end{itemize}

Each FSM is described entirely by its \texttt{arc} and initial \texttt{state} facts,
making the rules reusable across any FSM without modification.

\section{FSM 1 --- Soft Drink Vending Machine (Figure 3.5)}

\subsection*{State Diagram}

\begin{figure}[H]
  \centering
  \includegraphics[width=0.72\textwidth]{fig3_5.jpg}
  \caption{Figure 3.5 --- State Diagram for a Soft Drink Vending Machine
           accepting Quarters (Q) and Nickels (N)}
\end{figure}

The machine accepts coins until 25 cents have been deposited.
States represent accumulated credit: \texttt{start} (0\textcent),
\texttt{s5} (5\textcent), \texttt{s10} (10\textcent),
\texttt{s15} (15\textcent), \texttt{s20} (20\textcent), and \texttt{success} (25\textcent).

% ── Code ────────────────────────────────────────────────────────────────────
\subsection*{CLIPS Code --- \texttt{vending\_machine.clp}}

\subsubsection*{Templates and Rules (shared engine)}

\begin{lstlisting}[style=clipsStyle]
;;; Templates
(deftemplate state (slot name))
(deftemplate arc (slot from) (slot to) (slot input))

;;; Transition rule --- fires when a valid arc exists
(defrule fsm-transition
   ?current <- (state (name ?s))
   ?inp     <- (input ?value)
   (arc (from ?s) (to ?next) (input ?value))
   =>
   (retract ?current)          ; remove old state
   (retract ?inp)              ; consume the input fact
   (assert (state (name ?next)))
   (printout t "Transition: " ?s " --[" ?value "]--> " ?next crlf))

;;; Error rule --- fires when no arc matches
(defrule fsm-no-transition
   (state (name ?s))
   (input ?value)
   (not (arc (from ?s) (to ?) (input ?value)))
   =>
   (printout t "ERROR: No transition from " ?s
               " on input '" ?value "'" crlf))
\end{lstlisting}

\subsubsection*{Arc Facts}

\begin{lstlisting}[style=clipsStyle]
(deffacts vending-machine-arcs
   (arc (from start)   (to s5)      (input N))
   (arc (from start)   (to s10)     (input Q))
   (arc (from s5)      (to s10)     (input N))
   (arc (from s5)      (to s15)     (input Q))
   (arc (from s10)     (to s15)     (input N))
   (arc (from s10)     (to s20)     (input Q))
   (arc (from s15)     (to s20)     (input N))
   (arc (from s15)     (to success) (input Q))
   (arc (from s20)     (to success) (input N))
   (arc (from s20)     (to success) (input Q)))

(deffacts initial-state
   (state (name start)))
\end{lstlisting}

\subsubsection*{Key Design Decisions}

\begin{itemize}
  \item The input fact \texttt{(input ?value)} is \textbf{retracted} inside
        \texttt{fsm-transition}, preventing the rule from re-firing on the
        same input.
  \item Only \textbf{one} state fact exists at any time; the old one is
        retracted before the new one is asserted.
  \item The rules are \textbf{FSM-agnostic} --- only the arc facts change
        between machines.
\end{itemize}

\section{FSM 2 --- String Validator: WHILE, WRITE, BEGIN (Figure 3.6)}

\subsection*{State Diagram}

\begin{figure}[H]
  \centering
  \includegraphics[width=0.62\textwidth]{fig3_6.jpg}
  \caption{Figure 3.6 --- Part of a Finite State Machine for determining valid
           strings WHILE, WRITE, and BEGIN}
\end{figure}

The machine validates whether a sequence of character inputs spells one of the
three reserved keywords: \texttt{WHILE}, \texttt{WRITE}, or \texttt{BEGIN}.
Each state corresponds to a prefix of a valid keyword. Reaching \texttt{success}
means the full keyword was accepted.

% ── Code ────────────────────────────────────────────────────────────────────
\subsection*{CLIPS Code --- \texttt{string\_validator.clp}}

\subsubsection*{Templates and Rules}

The same generic engine rules (\texttt{fsm-transition} and
\texttt{fsm-no-transition}) are used --- only the arc facts differ.

\subsubsection*{Arc Facts}

\begin{lstlisting}[style=clipsStyle]
(deffacts string-validator-arcs
   ;; BEGIN path: B -> sB -> sBE -> sBEG -> sBEGI -> success
   (arc (from start) (to sB)      (input B))
   (arc (from sB)    (to sBE)     (input E))
   (arc (from sBE)   (to sBEG)    (input G))
   (arc (from sBEG)  (to sBEGI)   (input I))
   (arc (from sBEGI) (to success) (input N))

   ;; WHILE path: W -> sW -> sWH -> sWHI -> sWHIL -> success
   (arc (from start) (to sW)      (input W))
   (arc (from sW)    (to sWH)     (input H))
   (arc (from sWH)   (to sWHI)    (input I))
   (arc (from sWHI)  (to sWHIL)   (input L))
   (arc (from sWHIL) (to success) (input E))

   ;; WRITE path: W -> sW -> sWR -> sWRI -> sWRIT -> success
   (arc (from sW)    (to sWR)     (input R))
   (arc (from sWR)   (to sWRI)    (input I))
   (arc (from sWRI)  (to sWRIT)   (input T))
   (arc (from sWRIT) (to success) (input E)))

(deffacts initial-state
   (state (name start)))
\end{lstlisting}

\subsubsection*{Key Design Decisions}

\begin{itemize}
  \item \texttt{WHILE} and \texttt{WRITE} share the state \texttt{sW} after
        the first character; they diverge at the second character
        (\texttt{H} vs \texttt{R}).
  \item State names directly encode the prefix accepted so far
        (e.g.\ \texttt{sBEG} means ``B-E-G has been seen''), making the
        arc table easy to read and verify against the diagram.
  \item Any unexpected character leaves no matching arc, so
        \texttt{fsm-no-transition} fires and reports an error.
\end{itemize}
